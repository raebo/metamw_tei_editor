import excludedRoutes from "../../constants/excluded-routes";
import { useSelector } from "react-redux";
import { RootState } from "../../redux/redux.store";
import { getMe } from "../../services/user.service";

import React from 'react';
// import { useAuth } from './AuthContext.tsx.txt';

type GuardProps = {
  children: React.ReactNode;
};

export const Guard: React.FC<GuardProps> = ({ children } : GuardProps) => {
   const { isAuthenticated } = useAuth();

   if (!excludedRoutes.includes(window.location.pathname) && !isAuthenticated) {
     window.location.href = "/login"
     return null
   }

   return <>{children}</>;
};



const Guard = ( { children }: GuardProps) => {
  // const user = useSelector((state: RootState) => state.auth.user);
  const rootState = useSelector((state: RootState) => state)
  let user = rootState.auth.user

  let content

  if (!user) {
    try {
      // const userData = refreshUserData(rootState)
      getMe().then((data) => {
      // awaitRefreshUserState(rootState).then((data) => {
        if (data) { user = data }
      }).catch(err => {
        console.error('Error fetching user data:', err);
      })
    } catch (err) {
      console.error('Error fetching user data:', err);
    }
  }

  if (excludedRoutes.includes(window.location.pathname)) {
    content = children;
  } else if (user) {
    content = children;
  } else {
    window.location.href = "/login"
    return null
  }

  return <>{content}</>;

}

export default Guard;
